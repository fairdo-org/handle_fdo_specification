[S-4]: FDO Record Root Profile Validation

Input: List of FDO attributes a1=(a1,def,a1,val ), ..., an=(an,def,an,val).

Process: Check that there are at least the attributes `21.T11969/Type`, `21.T11969/Profile` and `21.T11969/Data` in the a1,def,...,an,def. 

enum Status {
    COMPLETE, REQUIRED_FIELDS, PROFILE_ONLY, TYPE_ONLY, NON_COMPLIANT
} // according to [S-4] in section 4 of the FDO Specifications

function validateAgainstRootProfile (fdo: FDORecord) -> Status {
    let status: Status = COMPLETE

    if (!fdo.containsAttribute('21.T11969/Profile')) {
        status = TYPE_ONLY // if 21.T11969/Profile is not found, at best the type can be found
    } else {
        if(!detectPID(typeValue)){
            status = TYPE_ONLY
        }
    }

    if (!fdo.containsAttribute('21.T11969/Type')) {
        if (status == TYPE_ONLY) {
            status = NON_COMPLIANT // Neither the type nor the profile has been found
        } else {
            status = PROFILE_ONLY // Only the profile has been found
        }
    } else {
        const typeValues = fdo.getAllValuesForAttribute('21.T11969/Type');
        for typeValue in typeValues {
            if (typeValue === "FDO_Profile" || typeValue === "FDO_Attribute_Definition" || typeValue === "FDO_Syntax_Definition" || typeValue === "FDO_Type_Definition"){
                continue; // This is the expected value for FDOs defining a profile, an attribute, a syntax definition, or an FDO type
            } else if (fdo.containsAttribute(typeValue)) {
                continue; // This is the expected case for attribute referencing
            } else if (detectPID(typeValue)){
                continue; // This is the expected case for profile referencing
            } else {
                if (status == PROFILE_ONLY) {
                    status = NON_COMPLIANT // 21.T11969/Profile was found, but the value of 21.T11969/Type is not compliant with the specification
                } else {
                    status = TYPE_ONLY // 21.T11969/Type was found, but the value of 21.T11969/Type is not compliant with the specification
                }
            }
        }
    }

    if (!fdo.containsAttribute("21.T11969/Data")){
        if (status == COMPLETE) {
            status = REQUIRED_FIELDS // Data has not been found, but profile and type were
        }
    } else {
        const dataValues = fdo.getAllValuesForAttribute("21.T11969/Data");
        boolean allValid = true;

        for dataValue in dataValues { // Inspect the values of all 21.T11969/Data attributes
            if (!fdo.containsAttribute(dataValue)){
                allValid = false // Check that every attribute reference actually exists in the record
            }
        }

        if (!allValid){ // Not all attribute references exist
            if (status == COMPLETE) {
                status = REQUIRED_FIELDS // The profile and type were found
            }
        }
    }

    return status
}

Output: One value from an enumeration of possible results

  • COMPLETE: 21.T11969/Profile, 21.T11969/Type, and 21.T11969/Data were found.
  • REQUIRED_FIELDS: 21.T11969/Profile and 21.T11969/Type were found. 21.T11969/Data is missing or invalid.
  • PROFILE_ONLY: 21.T11969/Profile has been found. 21.T11969/Type is missing or invalid. No assumption about 21.T11969/Data is made.
  • TYPE_ONLY: 21.T11969/Type has been found. 21.T11969/Profile is missing or invalid. No assumption about 21.T11969/Data is made.
  • NON_COMPLIANT: Neither 21.T11969/Profile nor 21.T11969/Type has been found.
