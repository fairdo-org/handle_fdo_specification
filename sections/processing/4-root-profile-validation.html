[S-4]: FDO Record Root Profile Validation

Input: List of FDO attributes a1=(a1,def,a1,val ), ..., an=(an,def,an,val).

Process: Check that there are at least the attributes `0.FDO/Type`, `0.FDO/Profile` and `0.FDO/Data` in the a1,def,...,an,def. 

enum Status {
    COMPLETE, REQUIRED_FIELDS, PROFILE_ONLY, TYPE_ONLY, NON_COMPLIANT
} // according to [S-4] in section 4 of the FDO Specifications

function validateAgainstRootProfile (fdo: FDORecord) -> Status {
    let status: Status = COMPLETE

    if (!fdo.containsAttribute('0.FDO/Profile')) {
        status = TYPE_ONLY // if 0.FDO/Profile is not found, at best the type can be found
    } else {
        if(!detectPID(typeValue)){
            status = TYPE_ONLY
        }
    }

    if (!fdo.containsAttribute('0.FDO/Type')) {
        if (status == TYPE_ONLY) {
            status = NON_COMPLIANT // Neither the type nor the profile has been found
        } else {
            status = PROFILE_ONLY // Only the profile has been found
        }
    } else {
        const typeValues = fdo.getAllValuesForAttribute('0.FDO/Type');
        for typeValue in typeValues {
            if (typeValue === "FDO_Profile" || typeValue === "FDO_Attribute_Definition" || typeValue === "FDO_Syntax_Definition" || typeValue === "FDO_Type_Definition"){
                continue; // This is the expected value for FDOs defining a profile, an attribute, a syntax definition, or an FDO type
            } else if (fdo.containsAttribute(typeValue)) {
                continue; // This is the expected case for attribute referencing
            } else if (detectPID(typeValue)){
                continue; // This is the expected case for profile referencing
            } else {
                if (status == PROFILE_ONLY) {
                    status = NON_COMPLIANT // 0.FDO/Profile was found, but the value of 0.FDO/Type is not compliant with the specification
                } else {
                    status = TYPE_ONLY // 0.FDO/Type was found, but the value of 0.FDO/Type is not compliant with the specification
                }
            }
        }
    }

    if (!fdo.containsAttribute("0.FDO/Data")){
        if (status == COMPLETE) {
            status = REQUIRED_FIELDS // Data has not been found, but profile and type were
        }
    } else {
        const dataValues = fdo.getAllValuesForAttribute("0.FDO/Data");
        boolean allValid = true;

        for dataValue in dataValues { // Inspect the values of all 0.FDO/Data attributes
            if (!fdo.containsAttribute(dataValue)){
                allValid = false // Check that every attribute reference actually exists in the record
            }
        }

        if (!allValid){ // Not all attribute references exist
            if (status == COMPLETE) {
                status = REQUIRED_FIELDS // The profile and type were found
            }
        }
    }

    return status
}

Output: One value from an enumeration of possible results

  • COMPLETE: 0.FDO/Profile, 0.FDO/Type, and 0.FDO/Data were found.
  • REQUIRED_FIELDS: 0.FDO/Profile and 0.FDO/Type were found. 0.FDO/Data is missing or invalid.
  • PROFILE_ONLY: 0.FDO/Profile has been found. 0.FDO/Type is missing or invalid. No assumption about 0.FDO/Data is made.
  • TYPE_ONLY: 0.FDO/Type has been found. 0.FDO/Profile is missing or invalid. No assumption about 0.FDO/Data is made.
  • NON_COMPLIANT: Neither 0.FDO/Profile nor 0.FDO/Type has been found.
